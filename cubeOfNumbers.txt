//Java program to print cube of numbers up to a particular integer value
 
package cube;
import java.util.*;
public class Cube {

    public static void main(String[] args) {
        int cube;
        Scanner sc=new Scanner(System.in);
        System.out.println("enter number"); // get the input number up to which the cubes have to be printed
        int n=sc.nextInt();
        for(int i=1;i<=n;i++){
            cube=i*i*i; // cube the number
            System.out.println("cube of " +i+ " is "+cube);
        }
            
    }
    
}
